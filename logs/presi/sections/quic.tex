\section{QUIC}
\begin{frame}[fragile]{Was ist QUIC?}
    QUIC will alles können was TCP kann nur besser
\end{frame}

\begin{frame}[fragile]{Was macht TCP aus?}
    \begin{itemize}
        \item Verbindungsorientiert
        \item Zuverlässige Übertragung
        \item Geordnete Zustellung
        \item Flusskontrolle
        \item Fehlerkontrolle
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{Grundlagen}
    \begin{itemize}
        \item Basiert auf UDP
        \item Stream Orientiert
        \item im Userspace
    \end{itemize}
\end{frame}
